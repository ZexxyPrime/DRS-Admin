Supabase Database
Go to supabase.com → Create free project → SQL Editor → Run this SQL to create all required tables:
CREATE TABLE leads ( id BIGSERIAL PRIMARY KEY, ref_id TEXT, full_name TEXT, email TEXT, whatsapp TEXT, alt_phone TEXT, service TEXT, sub_type TEXT, workers_required INTEGER, engagement_type TEXT, urgency TEXT, address TEXT, city TEXT, state TEXT, country TEXT DEFAULT 'India', pincode TEXT, start_date DATE, time_slot TEXT, duration TEXT, budget TEXT, special_skills TEXT, source TEXT, notes TEXT, admin_notes TEXT, status TEXT DEFAULT 'New', submitted_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE marketplace ( id BIGSERIAL PRIMARY KEY, ref_id TEXT, type TEXT, full_name TEXT, whatsapp TEXT, service TEXT, city TEXT, experience TEXT, rate TEXT, availability TEXT, description TEXT, documents TEXT, budget TEXT, urgency TEXT, admin_notes TEXT, status TEXT DEFAULT 'New', submitted_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE pageviews ( id BIGSERIAL PRIMARY KEY, page TEXT, referrer TEXT, is_new_session BOOLEAN DEFAULT true, visited_at TIMESTAMPTZ DEFAULT NOW() ); ALTER TABLE leads ENABLE ROW LEVEL SECURITY; ALTER TABLE marketplace ENABLE ROW LEVEL SECURITY; ALTER TABLE pageviews ENABLE ROW LEVEL SECURITY; CREATE POLICY "ins_leads" ON leads FOR INSERT WITH CHECK (true); CREATE POLICY "ins_mkt" ON marketplace FOR INSERT WITH CHECK (true); CREATE POLICY "ins_pv" ON pageviews FOR INSERT WITH CHECK (true); CREATE POLICY "all_leads" ON leads FOR ALL USING (true); CREATE POLICY "all_mkt" ON marketplace FOR ALL USING (true); CREATE POLICY "all_pv" ON pageviews FOR ALL USING (true);
